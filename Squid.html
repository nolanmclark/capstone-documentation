def squid_filter(event)
    entry = Struct.new(:epoch, :elapsed, :src_ip, :code, :status, :bytes, :method, :URL, :ident, :peerstat, :peerhost, :type, :auth, :squid_host)
    
    begin
        line = event.get("message")
        line = line.gsub(/\s+/m, " ").strip.split(" ")
    
        #logger.info "#{line}"

        code_stat = line[3].split("\/")
        line[3] = code_stat[0]
        line.insert(4, code_stat[1])

        peerstat_host = line[9].split("\/")
        line[9] = peerstat_host[0]
        line.insert(10, peerstat_host[1])
        
        sqdhost = event.get("host")
        sqdhost = sqdhost.split("\.")
        line.push(sqdhost[1])

        #logger.info "#{line}"

        out = entry.new(*line)
        out.each_pair {|k, v|
            if v == "-"
                event.set(k.to_s, nil)
            else
                event.set(k.to_s, v.to_s)
            end
        }

        return [event]
    rescue StandardError => e
        logger.error "#{line}"
        logger.error "#{e.message}"
        logger.error "#{e.backtrace.inspect}"
        return []
    end
end
