def test_FilterShouldSucceed_NoUserID()
    badgeLog = "badge 'end=1533317684', 'Union Pacific Railroad', 'NE_OMA_UPC_02_W_Lbby LowRise Turnstyle 4', 'Granted Access', '130578', 'Nicole, Dawn', 'Beasley', '0352313',"
    event = Hash.new("event")
    event = {"message" => badgeLog}
    response = filter(event)
    response = response[0]
    
    empId = response[:emp_id]
    userId = response[:user_id]

    assert_equal(empId, userId)
end
