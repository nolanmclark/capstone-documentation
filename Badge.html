def badge_filter(event)
    entry = Struct.new(:epoch, :location, :reader, :action, :badge_id, :fname, :lname, :emp_id, :user_id) {
        def initialize(*) super; self.user_id ||= self.emp_id; end
    }
    reg = /('[\w\s,\/\-\(\)\.&*À-ÿ]+'|"[\w\s,\/\-\(\)\.&*À-ÿ']+")/

    begin
        line = event.get("message")
        line = line.gsub(/^badge 'end=/, "'")
        line = line.gsub(/,$/, "")
        
        #logger.info "#{line}"

        m = line.scan(reg)
    
        out = entry.new(*m)
        out.each_pair {|k, v|
            out[k] = v[0][1..-2]
        }

        event.set('epoch', out[:epoch])
        event.set('location', out[:location])
        event.set('reader', out[:reader])
        event.set('action', out[:action])
        event.set('badge_id', out[:badge_id])
        event.set('emp_name', "#{out[:fname]} #{out[:lname]}".tr(',', ''))
        event.set('emp_id', out[:emp_id])
        event.set('user_id', out[:user_id])
        
        return [event]
    rescue
        logger.error "#{line}"
        return []
    end
end
